// Real-time collaborative coding example
function fibonacci(n) {
    if (n <= 1) return n;
    
    const cache = [0, 1];
    
    for (let i = 2; i <= n; i++) {
        cache[i] = cache[i - 1] + cache[i - 2];
    }
    
    return cache[n];
}

// Test the function
console.log('Fibonacci(10):', fibonacci(10));
// Output: 55
